# Lambda

```py
double = lambda x: x * 2
multiply = lambda x, y: x * y
add = lambda x, y, z: x + y + z
full_name = lambda first_name, last_name: first_name + " " + last_name
adult_age = lambda age: True if age >= 18 else False

print(double(2))
print(multiply(5, 6))
print(add(1, 2, 3))
print(full_name("Hello", "World!"))
print(adult_age(12))
```

# Sort

```py
students = [("WER", "F", 60),
            ("Qase", "A", 65)]

grade = lambda grades:grades[1]

students.sort(key=grade)

for i in students:
    print(i)
```

```py
students = (("WER", "F", 60),
            ("Qase", "A", 65))

grade = lambda grades:grades[1]

sorted_students = sorted(students, key=grade)

for i in sorted_students:
    print(i)
```

# Map

Applies a function to each item in an iterable (list, tuple, etc.)

```py
store = [("shirt", 20.00),
         ("pants", 35.00),
         ("jackets", 50.00),
         ("socks", 10.00)]

to_euros = lambda data: (data[0], data[1] * 0.82)
to_dollars = lambda data: (data[0], data[1] / 0.82)

store_euros = list(map(to_euros, store))
store_dollars = list(map(to_dollars, store))

for i in store_euros:
    print(i)

for j in store_dollars:
    print(j)
```

# Filter

```py
friends = [("Qeir", 20),
           ("Mocne", 18),
           ("Peorj", 23),
           ("Joey", 16),
           ("Weri", 21)]

age = lambda data:data[1] >= 18

adult = list(filter(age, friends))

for i in adult:
    print(i)
```
